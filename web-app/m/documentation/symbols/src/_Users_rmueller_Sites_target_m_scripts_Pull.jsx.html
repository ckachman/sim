 65 		var curField = m.initTextField(data.config.fieldID);
 66 		curField.val('');

 75 		u.setMSecs();
 76 		return m.doAjaxFormSubmit(form);
 77 	},
 78 	PULL006_success: function (tableData) {
 79 		u.getMSecsData();
 80 		if (tableData.status == 'success') {
 81 			m.curApp.resetApp(); // start the app over from page 1 now
 82 		} else {
 83 			m.popError(tableData.message, 'error');
 84 		}
 85 	}
