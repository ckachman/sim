  8 This gets the MAC address and sets the proper CE variables
  9 
 10 @function GetMACAddressOfDevice
 11 */
 12 function GetMACAddressOfDevice() {
 13 	var macAddr = "Unknown", device = document.getElementById("txtDeviceID");
 14 	if (navigator.platform.indexOf("WinCE") >= 0) {
 15 		macAddr = external.CETerm.GetProperty("device.macaddress");
 16 	}
 17 	device.value = macAddr;
 18 }

 66 u.timeNowTimer = [];
 67 u.initStopwatch = function () {
 68 	var ticks = 0;
 69 	if (navigator.platform.indexOf("WinCE") >= 0) {
 70 		ticks = (external.OS.TickCount);
 71 	} else {
 72 		ticks = (new Date().getTime());
 73 	}
 74 	return ticks;
 75 };
 76 u.timeNow = u.initStopwatch();

 78 u.setMSecs = function () {
 79 	u.timeNow = u.initStopwatch();
 80 };
 81 u.getMSecs = function () {
 82 	var nowSecs = u.initStopwatch(), curMSecs = nowSecs - u.timeNow, color = '#006633';
 83 	if (curMSecs > 100) { color = '#ff6600'; }
 84 	if (curMSecs > 500) { color = '#cc3300'; }
 85 	if (curMSecs > 1000) { color = '#cc0000'; }
 86 	$('#time').show().css({'color': color});
 87 	$('#time').append([(curMSecs), " msecs | "].join(''));
 88 	clearTimeout(u.timer);
 89 	u.timer = setTimeout(u.clearTimeHtml, 2000);
 90 	return curMSecs;
 91 };
 92 u.clearTimeHtml = function () {
 93 	$('#time').html('');
 94 };

104 	$('#timeData').show().css({'opacity': '1', 'color': color});
105 	$('#timeData').append([(curMSecs), " msecs | "].join(''));

138 u.getMSecsTimer = function (timer, msg, keepRunning) {
139 	if (u.debug) {
140 		var nowSecs = u.initStopwatch(), curMSecs = nowSecs - u.timeNowTimer[timer - 1];
141 		$('#timeTimer').append([msg, ': ', (curMSecs), " msecs<br>"].join(''));
142 		$('#timeTimerContainer').scrollTop($('#timeTimer').height());
143 		if (!keepRunning) {u.timeNowTimer[timer - 1] = null; }
144 		return curMSecs;
145 	}

175 // base Screen Load event - offers beep functionality
176 u.baseLoadEvent = function () {
177 	try {
178 		// For Beep(s) on Windows CE
179 		if (navigator.platform.indexOf("WinCE") >= 0) {
180 			u.beepCounter = 0;
181 			u.beeper();
182 		}
183 	} catch (e) {
184 		alert(e.message);
185 	}
186 };
187 
188 
189 u.beeper = function () {
190 	external.OS.PlayTone(99, 100, 50);
191 	//external.CETerm.PlayTone( 99, 100, 50 );
192 	if (u.beepCount > u.beepCounter + 1) {
193 		u.beepCounter++;
194 		setTimeout(u.beeper, 10);
195 	}
196 };

200 };

203 // Scan Config event
204 u.baseScanConfigEvent = function () {
205 	try {
206 		// First Beep on Windows CE
207 		if (navigator.platform.indexOf("WinCE") >= 0) {
208 			var sessIdx = external.sessionindex;
209 			//enable symbologies
210 			external.CETerm.SetProperty(["session",  sessIdx,  ".scanner.",  "upca",  ".enabled"].join(''), true);
211 			external.CETerm.SetProperty(["session",  sessIdx,  ".scanner.",  "upce0",  ".enabled"].join(''), true);
212 			external.CETerm.SetProperty(["session",  sessIdx,  ".scanner.",  "ean13",  ".enabled"].join(''), true);
213 			external.CETerm.SetProperty(["session",  sessIdx,  ".scanner.",  "ean18",  ".enabled"].join(''), true);
214 			// Set the decode beepTime
215 			external.CETerm.setProperty(["session", sessIdx, ".scanner.decodebeeptime"].join(''), u.beepTime);
216 			// Apply the settings -- why are these now broken?
217 			//external.CETerm.GETIDA("IDA_SCAN_RESUME", 0);
218 			//external.CETerm.GETIDA("IDA_SCAN_APPLYCONFIG", 0);
219 		}
220 	} catch (e) {
221 		alert(e.message);
222 	}
223 };

238 // Scan Capture event
239 u.baseScanEvent = function (data, source, type, time, length) {
240 	if (!u.isSubmitting) {
241 		m.setScanData(data);
242 		u.isSubmitting = true;
243 		m.submitCurrentForm();
244 		external.CETerm.GETIDA("IDA_SCAN_SUSPEND", 0); //turn off scanner, user experience, limit to one scan per submit
245 	}
246 };
247 
248 u.onScanEvent = function scanEvent(data, source, type, time, length) {
249 	u.baseScanEvent(data, source, type, time, length);
250 };
251 
252 u.onscan = function (data, source, type, time, length) {
253 	//u.logStats("OnScanSubmit ");
254 	var inputTypeElem = document.getElementById('txtInputType'), barcodeTypeElem = document.getElementById('txtBarcodeType'), btArray = source.split(':');
255 	inputTypeElem.value = 'S';
256 	if (btArray.length > 1) { barcodeTypeElem.value = btArray[1]; }
257 	u.onScanEvent(data, source, type, time, length);
258 };

269 u.sip_hide = function () {
270 	// check if CE device
271 	if (navigator.appVersion.indexOf("Windows CE") >=  0) {

276 u.sip_show = function () {
277 	// check if CE device
278 	if (navigator.appVersion.indexOf("Windows CE") >=  0) {
279 		external.CETerm.GETIDA("IDA_SIP_SHOW", 0);
280 		return false;
281 	}
282 	return false;
283 };
284 
285 u.sip_toggle = function () {
286 	if (u.sipState == false) {
287 		external.CETerm.GETIDA("IDA_SIP_SHOW", 0);
288 		u.sipState = true;
289 		return false;
290 	} else {
291 		external.CETerm.GETIDA("IDA_SIP_HIDE", 0);
292 		u.sipState = false;
293 	}
294 };
