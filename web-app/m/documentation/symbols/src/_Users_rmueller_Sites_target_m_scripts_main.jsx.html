 14 Clicking on a subapp button calls that subapp's js file and creates a new subapp object 
 15 unless one already exists in which case the subapp's toggleIn() function is called. The 
 16 subapp is responsible for loading itself.
 17 
 18 When the toggle (or currently, home as well) button is clicked, the current subapp's 
 19 toggleOut() function is called, which will save its current state to the server (or 
 20 retaining itself in browser memory). The main app hides the subapp.
 21 
 22 Be careful when adding event handlers in the subApp page functions. These get called 
 23 every time the app toggles, so if you add a button click without removing previous 
 24 button clicks first (using button.unbind('click') for example), that button click 
 25 event fires multiple times.

194 	var data = dataToPass || '';

273 Wrapper function for forcing numeric values in a field on key press
274 
275 @function m.forceNumericOnly

281 /**
282 Forcing numeric values only. Not called directly.
283 
284 @function m.forceNumericOnlyHandler
285 @param {event} e The event contains the key pressed.
286 */
287 m.forceNumericOnlyHandler = function (e) {
288 	var key = e.charCode || e.keyCode || 0, fail = false;
289 	// allow backspace, tab, delete, arrows, numbers and keypad numbers ONLY
290 	if (key === 8 || key === 9 || key === 46 || key === 13 || key === 35 || key === 36 || (key >= 37 && key <= 40)) {
291 		fail = false;
292 	} else if (!(e.shiftKey) && (key >= 48 && key <= 57)) {
293 		fail = false;
294 	} else if ((key >= 96 && key <= 105)) {
295 		fail = false;
296 	} else { fail = true; }
297 	if (fail) { e.preventDefault(); }
298 };

315 m.forceAlphaNumericOnlyHandler = function (e) {

419 /**
420 Saves the current app's state and switches to toApp. Creates toApp if it isn't already.
421 
422 @function m.toggleApps
423 @param {string} toApp The app to switch to ('Backstock').
424 */
425 m.toggleApps = function (toApp) {
426 	var tatimer = u.setMSecsTimer();
427 	// save the current app's state to the server and load the next app's state from the server
428 	if (typeof m.curApp != 'undefined') { m.curApp.toggleOut(); }
429 	if (!m[toApp]) { // if the toApp doesn't exist yet, create it
430 		m[toApp] = m.createSubApp(toApp);
431 	} else { // otherwise, just toggle back to it
432 		m[toApp].toggleIn(toApp);
433 	}
434 	// activate the current app. The init() or toggleIn() function of the app will activate that app's first page, so flow switches to the app's .js file.
435 	$('.appfrag.active').removeClass('active');
436 	$(['#', toApp].join('')).addClass('active');
437 	u.getMSecsTimer(tatimer, ['m.toggleApps ', toApp, ' end'].join(''));
438 };

487 /**
488 Creates a grid table in the div using the data.header[] and data.data[] arrays.
489 
490 @function m.createGridTable
491 @param {object} div The div in which to place the .GridView
492 @param {object} data The data with which to populate the .GridView
493 @param {number} numRows The number of rows to show before pagination occurs
494 */
495 m.createGridTable = function (div, data, numRows) {
496 	//var tmr = u.setMSecsTimer('m.createGridTable start');
497 	// set up pagination (only applies to this page!)
498 	data.gridTable.tableheight = data.gridTable.data.length;
499 	data.gridTable.numpagesshowing = numRows || 3;
500 	data.gridTable.numpages = Math.ceil(data.gridTable.tableheight / data.gridTable.numpagesshowing);
501 	data.gridTable.curpage = 0;
502 	if ($('.GridView', div).length == 0) { // if we've already processed this grid, don't process it again!
503 		// first, copy the tableDataGridView code fragment into the page
504 		div.html($('#codeFragments #tableDataGridView').html());
505 		//var divName = div.closest('.pagefrag').attr('id');
506 		if (data.gridTable.tableTitle) {
507 			$('.tableTitle', div).html(data.gridTable.tableTitle).show();
508 		} else {
509 			$('.tableTitle', div).hide();
510 		}
511 		if (data.gridTable != null) {
512 			// load in the header strings 
513 			var table_headers = [], len = data.gridTable.header.length - 1, table_data = [], show = data.gridTable.numpagesshowing, i, j, d;
514 			for (i = 0; i <= len; i++) {
515 				table_headers.push(["<th scope='col'>", data.gridTable.header[i], "</th>"].join(''));
516 			}
517 			$("thead tr", div).html(table_headers.join(""));
518 			// now load in the table data itself
519 			for (i = 0; i < show; i++) {
520 				table_data.push("<tr>");
521 				d = data.gridTable.data[i];
522 				for (j in d) {
523 					if (d.hasOwnProperty(j)) {
524 						table_data.push(["<td class='disabled'>", d[j], "</td>"].join(""));
525 					}
526 				}
527 				table_data.push("</tr>");
528 			}
529 			$("tbody", div).html(table_data.join(""));
530 			// check to see if we need pagination
531 			if (data.gridTable.tableheight > data.gridTable.numpagesshowing + 1) {
532 				div.append('<div class="gridFooter"></div>');
533 				m.appendPaginationButtons();
534 			}
535 			m.gridViewPage(0);
536 		} else {
537 			div.html(data.locationMessage);
538 		}
539 	} else {
540 		// just show app nav buttons if they exist, and set the curpage to the first page
541 		if (data.gridTable.tableheight > data.gridTable.numpagesshowing + 1) { m.appendPaginationButtons(); }
542 		data.gridTable.curpage = 0;
543 		m.gridViewPage(0);
544 	}
545 };

575 /**
576 Click handler for .GridView. Where dir is 1 it shows the next page, where dir is -1 it shows the previous page.
577 
578 @function m.gridViewPage
579 @param {number} dir The direction of pagination; -1 for up, 1 for down.
580 */
581 m.gridViewPage = function (dir) {
582 	//var gvp = u.setMSecsTimer('m.gridViewPage start')
583 	u.setMSecs();
584 	var data = m.curApp.appStack.pageDataLoad[m.curApp.appStack.pageDataLoad.length - 1].gridTable, curGrid = $(['#', m.curApp.curPage, ' .GridView'].join('')), firstpos, count, fsp, dhl, i, k;
585 	if ((dir == 1 && data.curpage + 1 < data.numpages) || (dir == -1 && data.curpage > 0) || dir == 0) {
586 		if (dir != 0) { // this means we're not resetting the grid view to the first page
587 			data.curpage += dir;
588 		} else {
589 			data.curpage = 0;
590 		}
591 		firstpos = data.curpage * data.numpagesshowing;
592 		count = 0;
593 		fsp = firstpos + data.numpagesshowing;
594 		dhl = data.header.length;
595 		// write the data into the table td's
596 		for (i = firstpos; i < fsp; i++) {
597 			for (k = 0; k < dhl; k++) {

601 					$($('td', $('tr', curGrid)[count + 1])[k]).html('');
602 				}
603 			}
604 			count++;
605 		}
606 	}
607 	// write out the pagination line ('1 of 5')
608 	$(".gridFooter", $(['#', m.curApp.curPage].join(''))).html([m.curApp.strings.page, ' ', data.curpage + 1, ' ', m.curApp.strings.of, ' ', data.numpages].join(""));

718 		var curP = m.curApp.appStack.curPage[0], curD = m.curApp.appStack.pageDataLoad[0];

728 		var url = '', frmMenu = '';

784 	/**
